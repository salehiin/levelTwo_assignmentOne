class Vehicle {
    make: string;
    year: number;
    constructor(make: string, year: number){
        this.make = make;
        this.year = year;
    }
    getInfo(){
        console.log(`This car made by ${this.make} in ${this.year}`)
    }
}

class Car extends Vehicle {
   
    model: string;
    constructor(make: string, year: number, model: string){
        super(make, year)
        this.model = model;
    }
    
    getModel(){
        console.log(`The car model is ${this.model} made by ${this.make} in ${this.year}`)
    }
}
